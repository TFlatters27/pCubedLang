if (strcmp(parser->current_token->value, "SUBROUTINE") == 0)
  {
    printf("defining subroutine\n");
    return parser_parse_subroutine_definition(parser, scope);
  }

  if (lexer_next(parser->lexer)->type == TOKEN_ASSIGNMENT)
  {
    printf("defining variable\n");

    return parser_parse_variable_definition(parser, scope);
  }
  else
  {
    return parser_parse_variable(parser, scope);
  }
